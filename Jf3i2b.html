코드 빈칸 채우기

import java.util.Scanner; 
 
 ThreeCupGame										 
{ 
	  int GAME_OVER = -1;		 
 
	 String strCups;							 
	 String strBall;							 
	 String strBlanks;						 
	 
	 String strMeet;						 
	 String strDodge;					 
	 String message;						 
 
	 ThreeCupGame() 						 
	{		 
		.strCups   = "  ___    ___    ___  \n"		 
				           + " |   |  |   |  |   | \n" 
				           + " |   |  |   |  |   | \n"; 
		.strBall   = "   ●   ";										 
		.strBlanks = "       ";										 
		.strMeet   = " 찾았다! ";										 
		.strDodge  = " 놓쳤다! ";										 
		 
		.message = " 1, 2, 3 중에서 구슬[●]을 숨긴 컵을 찾으세요 : ";	 
	}  
 
	 String play( Scanner scan )				 
	{	 
		int ball =  
		           selectCup(); 
		int player =  
		             selectCup( scan ); 
		String result =   
		                getResult( player, ball ); 
		return result;	 
	} 
 
	 int selectCup( Scanner scan ) 
	{ 
		int player = 0; 
		try 
		{ 
			System.out.print( message ); 
			player =  
			         scan.nextInt(); 
			if ( isPossibleValue( player - 1 )  
			                                   == true ) 
				player = player - 1; 
			else 
				player = GAME_OVER; 
		} 
		catch( Exception e ) 
		{ 
			e.printStackTrace(); 
		} 
 
		return player; 
	} 
 
	 int selectCup()  
	{  
		return (int)( Math.random() * 3 ); 
	} 
	 
	 boolean isPossibleValue( int selection ) 
	{ 
		return ( 0 <= selection ) && ( selection <= 2 );		 
	} 
 
 
	 String getResult( int player, int ball )  
	{  
		if ( player == GAME_OVER ) 
			return null; 
		 
		String str = strCups; 
		for ( int i = 0; i < 3; i++ ) 
		{ 
			if ( ball == i )  
				str += strBall; 
			else 
				str += strBlanks; 
		} 
		 
		str += "\n"; 
		for ( int i = 0; i < 3; i++ ) 
		{ 
			if ( ( player == i ) && ( player == ball ) ) 
				str += strMeet; 
			else if ( player == i ) 
				str += strDodge; 
			else 
				str += strBlanks; 
		} 
		 
		return str; 
	} 
} 
 
		 
  ThreeCupGame_Main										 
{ 
	 static void main(String[] args)  
	{ 
		Scanner scan = new Scanner( System.in ); 
		ThreeCupGame game = 
		                      new ThreeCupGame(); 
			 
		for( String result = null;  
		                          ( result =  
		                                       game.play( scan ) 
		                                                         ) != null; ) 
		{ 
			System.out.println( result );  
		} 
		 
		System.out.println( "\n게임을 종료합니다. 다음에 또 만나요!" );  
	} 
}
