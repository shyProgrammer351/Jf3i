import java.util.Scanner; 
	 
 Calendar   
{ 
	 boolean IsLeapYear( int year )										 
	{ 
		if ( ( year % 400 ) == 0 ) 
			 true;										 
		else if ( ( year % 100 ) == 0 ) 
			 false;										 
		else if ( ( year % 4 ) == 0 ) 
			 true;										 
		else  
			 false;										 
	} 
	 
	 int getLastDay( int month, int year )									 
	{ 
		final int lastDayPerMonth[] = { 0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };   
	 
		if ( IsLeapYear( year )   
		                        && ( month == 2 ) ) 
			 lastDayPerMonth[ month ] + 1;							 
		else 
			 lastDayPerMonth[ month ]; 							 
	} 
				 
	 int getFirstDay( int month, int year )							 
	{ 
		int totalDays = (year-1)*365 + (year-1)/4 - (year-1)/100 + (year-1)/400; 
		for ( int idx = 1; idx < month; idx++ ) 
			totalDays += getLastDay( idx, year ); 
		 ( ( totalDays + 1 ) % 7 );									 
	} 
	 
	 void print( int year, int month, int firstDay, int lastDay )				 
	{ 
		int day = 0; 
		int dayOfWeek = 0; 
		final int sunday = 0;		 
		final int saturday = 6;		 
	 
		System.out.println( "\n              " + year + "년 " + month + "월\n" ); 
	 
		for( dayOfWeek = sunday; dayOfWeek < firstDay; dayOfWeek ++ ) 
		{ 
			System.out.print( "     " ); 
		} 
	 
		for( day = 1; day <= lastDay; dayOfWeek++, day++ ) 
		{ 
			System.out.printf( "%5d", day ); 
			 
			if ( dayOfWeek == saturday ) 
			{ 
				System.out.println(); 
				dayOfWeek = sunday - 1; 
			} 
		} 
	} 
}

	 static void main(String args[])  
	{  
		Scanner scan = new Scanner( System.in ); 
		Calendar calendar = new Calendar(); 
		int year  = 0; 
		int month  = 0; 
		int lastDay= 0; 
		int firstDay = 0; 
	 
		System.out.print( "년도를 입력하세요: " ); 
		year = scan.nextInt(); 
		System.out.print( "월을 입력하세요: " ); 
		month = scan.nextInt(); 
		 
		firstDay = 									  
		          calendar.getFirstDay( month, year ); 
		lastDay = 										  
		          calendar.getLastDay( month, year ); 
	 
		calendar.print( year, month, firstDay, lastDay ); 
	 
	}
