import java.util.Scanner; 
	 
 Calendar   
{ 
	final int lastDayPerMonth[] = { 0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };   
	 
	 boolean IsLeapYear( int year )										 
	{ 
		if ( ( year % 400 ) == 0 ) 
		{ 
			 true;										 
		} 
		else if ( ( year % 100 ) == 0 ) 
		{ 
			 false;										 
		} 
		else if ( ( year % 4 ) == 0 ) 
		{ 
			 true;										 
		} 
		else  
		{ 
			 false;										 
		} 
	} 
	 
	 int getLastDay( int month, int year )									 
	{ 
		if ( ( month == 2 ) &&  
		                         IsLeapYear( year ) ) 
		{ 
			 lastDayPerMonth[ month ] + 1;							 
		} 
		else 
		{ 
			 lastDayPerMonth[ month ]; 							 
		} 
	} 
			 
	 int getTotalDaysBetweenNewYearsDayAnd( int previousMonth, int thisYear )				 
	{ 
		int totalDays = 0;  
		int month = 0; 
		 
		for ( month = 1; month <= previousMonth; month++ ) 
		{ 
			totalDays += lastDayPerMonth[ month ]; 
		} 
		if ( ( month >= 2 ) && IsLeapYear( thisYear ) ) 
		{ 
			totalDays++; 
		} 
 
		 totalDays;										 
	} 
	 
	 int getFirstDay( int month, int year )							 
	{ 
		int totalDays = (year-1)*365 + (year-1)/4 - (year-1)/100 + (year-1)/400; 
		totalDays +=  					  
		            getTotalDaysBetweenNewYearsDayAnd( month-1, year );	 
		 ( (totalDays+1) % 7 );									 
	} 
	 
	 void print( int year, int month, int firstDay, int lastDay )				 
	{ 
		int day = 0; 
		int dayOfWeek = 0; 
		final int sunday = 0;		 
		final int saturday = 6;		 
	 
		System.out.println( "\n              " + year + "년 " + month + "월\n" ); 
	 
		for( dayOfWeek = sunday; dayOfWeek < firstDay; dayOfWeek ++ ) 
		{ 
			System.out.print( "     " ); 
		} 
	 
		for( day = 1; day <= lastDay; dayOfWeek++, day++ ) 
		{ 
			System.out.printf( "%5d", day ); 
			 
			if ( dayOfWeek == saturday ) 
			{ 
				System.out.println(); 
				dayOfWeek = sunday - 1; 
			} 
		} 
	} 
}

		firstDay = 									  
		          calendar.getFirstDay( month, year );
