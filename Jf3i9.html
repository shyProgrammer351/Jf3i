알고리즘

 프로그램시작 
    하노이 타워 생성 
    3개의 원판을 1(출발), 2(임시), 3(도착) 기둥을 이용해 이동 
 프로그램종료 
 
 원판 이동 함수 시작 
    N-1개 원판을 출발기둥에서 임시기둥으로 모두 이동 
    가장 큰 원판을 출발기둥에서 도착기둥으로 이동 
    N-1개 원판을 임시기둥에서 도착기둥으로 모두 이동 
 원판 이동 함수 종료

	// 파일명 : ./Chapter03/TowerOfHanoi_Main.java 

 

	class TowerOfHanoi 

	{ 

		// 원판 이동 함수 시작 

1		public void moveTower( int numDisks, int start, int temp, int end ) 

		{ 

2			if (numDisks == 1) 

3				moveOneDisk (start, end);	 

4			else 

			{ 

		  		// N-1개 원판을 출발기둥에서 임시기둥으로 모두 이동 

5				moveTower( numDisks-1, start, end, temp );	 

		  		// 가장 큰 원판을 출발기둥에서 도착기둥으로 이동 

6				moveOneDisk (start, end);

7				moveTower( numDisks-1, temp, start, end );	 

			} 

		// 원판 이동 함수 종료 

8		} 

 

9	   private void moveOneDisk (int start, int end ) 

	   { 

10	      System.out.println ("Move one disk from " + start + " to " + end);
