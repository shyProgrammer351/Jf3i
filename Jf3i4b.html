코드 빈칸 채우기

import java.util.ArrayList; 
import java.util.Scanner; 
 
 Chatbot								 
{ 
	  pairs = 						 
				{ { "이름이 뭐", "난 재석이야. 국민MC 유재석과 이름이 같아. ㅎㅎㅎ 넌 몇 살이니?" }, 
				  { "몇 살", "난 20살이야. 남들이 다 부러워하는 나이지. 넌 어디 살아?" }, 
				  { "어디 살", "난 홍지동 살아. 북한산 아래 있어서 공기가 무지 맑아.ㅎㅎㅎ 취미가 뭐니?" }, 
				  { "만나서 반가", "나도 반가워 친하게 지내자" } }; 
	 
	 Chatbot()					 
	{ 
	} 
 
	 String getResponse( String userUtterance )  
	{ 
		if ( userUtterance == null ) 
		{ 
			 "(시스템) 안녕? 넌 이름이 모야?";								 
		}		 
		else if ( userUtterance.( "바이" ) == true )  
		{ 
			 null;																	 
		} 
 
		for ( String[] pair : pairs ) 
		{ 
			if ( userUtterance.( pair[0] ) == true )  
			{ 
				 "(시스템) " + pair[1]; 									 
			} 
		} 
 
		 "(시스템) 뭐 딴거 재밌는거 없을까?";		 
	} 
 
	 String getFinalResponse() 
	{ 
		 "(시스템) 그래 안녕! 담에 또 봐!";											 
	} 
} 
 
  Chatbot_Main								 
{ 
	 static void main(String[] args) 	 
	{ 
		Scanner scan =  Scanner( System.in );  
		 
		Chatbot chatbot =  
		                   Chatbot();	 
 
		for ( String input = null, output = null; 
		                                          ( output =  
		                                                     chatbot.getResponse( input ) )  
		                                                                                    != null; ) 
		{ 
			System.out.print( output + "\n 사용자: " ); 
			input =  
			        scan.nextLine(); 
		} 
 
		System.out.print(  
		                  chatbot.getFinalResponse() + "\n" ); 
		scan.close(); 
			 
	} 
}
